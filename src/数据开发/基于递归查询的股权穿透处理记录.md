## 基于递归查询的股权穿透处理记录

- 根据股东底表进行20层股权穿透
- 去除股权循环情况

```sql
with recursive tmp(最底层企业名称,穿透认缴出资比例,企业名称,投资人名称,深度,投资人路径) as (
    
select 企业名称 最底层企业名称,认缴出资比例 穿透认缴出资比例,企业名称,投资人名称
    ,1 深度,投资人名称 投资人路径
from 股东底表
    
union all
    
select t1.最底层企业名称,t1.穿透认缴出资比例*t2.认缴出资比例 穿透认缴出资比例,t2.企业名称,t2.投资人名称
    ,深度+1 深度,concat(投资人路径,',',t2.投资人名称) 投资人路径
from tmp t1
join 股东底表 t2
on t1.投资人名称=t2.企业名称 and find_in_set(t2.投资人名称,t1.投资人路径)=0
    
where 深度<20
    
)
select * from tmp
```
